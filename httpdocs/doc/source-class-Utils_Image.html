 34     static function writeText($image,$xc,$yc,$text,$fontColor,$arOptions=array())
 35     {
 36     if (empty($text)) $text= '________';
 37     $fullFont=$_SERVER['DOCUMENT_ROOT'].'/lib/fonts/'.self::$font;
 38     $fontSize= (isset($arOptions['FONTSIZE']) ? $arOptions['FONTSIZE']: self::$defaultFontSize);
 39     $arSizes = imagettfbbox($fontSize, 0, $fullFont, $text);
 40     $texWidht=$arSizes[2]-$arSizes[0];
 41     $textHeight=$arSizes[5]-$arSizes[1];
 42     $x=self::calcX($xc - $texWidht/2);
 43     $y=self::calcY($yc + ($arSizes[5]-$arSizes[1])/2);
 44     if ($x < 0) $x=0;
 45     if ($x+$texWidht > self::$pageWidth) $x=self::$pageWidth-$texWidht;
 46     if ($y-$textHeight < 0) $y=$textHeight;
 47     if ($y > self::$pageHeight) $y=self::$pageHeight;
 48     if (isset($arOptions['OFFSET_YZERO']) && intval($yc)==0) $y-=$arOptions['OFFSET_YZERO'];
 49     imagettftext($image, $fontSize,0,$x,$y,$fontColor,$fullFont,$text);
 50     }

 53     static function getImage($arData)
 54     {
 55     $fullFont=$_SERVER['DOCUMENT_ROOT'].'/lib/fonts/'.self::$font;
 56     $countGroup = count($arData['CATEGORIES']);
 57     $image = @imagecreate(self::$pageWidth, self::$pageHeight);
 58     $backgroung_color = imagecolorallocate($image,self::$colorBackG[0],self::$colorBackG[1],self::$colorBackG[2]);
 59     $text_color = imagecolorallocate($image, 0,0,0);
 60     self::$ellipseWidth = (self::$pageWidth/2-self::$margin) * self::$k;
 61     self::$ellipseHeight = (self::$pageHeight/2-self::$margin) * self::$k;
 62     imageellipse($image,  self::calcX(0), self::calcY(0), self::$ellipseWidth, self::$ellipseHeight, $text_color);
 63     $angle = 360/$countGroup;
 64     self::writeText($image, 0, 0, $arData['ANSWER']['DATA']['OBJECT'], $text_color);
 65     $i=0;
 66     foreach($arData['CATEGORIES'] as $categoryId=>$arCategory)
 67     {
 68         $x= (self::$pageWidth/2-self::$margin)*cos(deg2rad($i*$angle));
 69         $y= (self::$pageHeight/2-self::$margin)* sin(deg2rad($i*$angle));
 70         $xm= (self::$pageWidth/2-self::$margin+self::$roundBig/2)*cos(deg2rad($i*$angle));
 71         $ym= (self::$pageHeight/2-self::$margin+self::$roundBig/2)*sin(deg2rad($i*$angle));
 72         $x0= (self::$ellipseWidth/2)*cos(deg2rad($i*$angle));
 73         $y0= (self::$ellipseHeight/2)* sin(deg2rad($i*$angle));
 74         imageline($image,  self::calcX($x0),  self::calcY($y0),  self::calcX($x),self::calcY($y), $text_color);
 75         $countElements=count($arCategory['ELEMENTS']);
 76         $deltaX=($x-$x0)/($countElements+1);
 77         imageellipse($image, self::calcX($xm),self::calcY($ym), self::$roundBig, self::$roundBig, $text_color);
 78         self::writeText(
 79             $image,
 80             $xm,
 81             $ym,
 82             $arCategory['NAME'],
 83             $text_color,
 84             array(
 85             'OFFSET_YZERO'=>self::$roundBig,
 86             'FONTSIZE'=>  self::$groupTextSize
 87             )
 88          );
 89         for($u=1;$u<=$countElements;++$u)
 90         {
 91         $coordX = $x0 + $u * $deltaX;
 92         $coordY = (($y-$y0)/($x-$x0))*($coordX-$x0)+$y0;
 93         imageellipse($image,self::calcX($coordX),self::calcY($coordY),self::$roundSmall, self::$roundSmall, $text_color);
 94         $text=$arData['ANSWER']['DATA']['ELEMENTS'][$arCategory['ELEMENTS'][$u-1]['ID']];
 95         self::writeText($image, $coordX, $coordY, $text, $text_color,array(
 96             'OFFSET_YZERO'=>self::$roundSmall*2,
 97             'FONTSIZE'=>  self::$elementsTextSize
 98             ));
 99         }
100         $i++;
101     }
102     Header('Content-type: image/png');
103     imagepng($image);

105     }
